// TODO : funcion que le pasas un numero y muestra por pantalla todos los numeros desde el numero hasta cero y al lado tantos guiones como valor del numero ej: f(3) 3 ---  
//                                                                                                                                                                   2 -- 
//																																							 1 - 
//																																							 0 
function muestraNumerosDescententeACeroConGuiones(a){
    for(var i=a;i>=0;i--){
		var cadena = "";
		for(var j=0;j<i;j++){
			cadena = cadena + "-";
		}
		var temp = i.toString();
		temp = temp + " " + cadena;
		print(temp);
	}
}
// TODO : funcion que recibe un numero y pinta un cuadrado del tamaÃ±o del lado igual al numero ej : f(2)   - -
//                                                                                                        |   |
//	                                																	  |   |
//									         															   - -
function pintaCuadrado(a){
	var cadena0 = "";
	var cadena1 = " -";
    for(var i=0;i<a;i++){
		cadena0=cadena0+cadena1;
	}
	cadena2="";
	cadena3="| ";
    for(var i=1;i<a;i++){
		cadena2=cadena2+"  ";
	}
	cadena3=cadena3+cadena2+"|";
	print(cadena0);
    for(var i=0;i<a;i++){
		print(cadena3);
	}
	print(cadena0);
}
